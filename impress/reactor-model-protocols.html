  sends: Events[T], out: Channel[(T, Long)],
  acks: Events[Long]) {
  var lastStamp, lastAck = 0L
  def canSend = lastStamp - lastAck < window
  val queue = new Queue[T]

    if (canSend) {
      lastStamp += 1
      out ! (x, lastStamp)

  acks onEvent { stamp =>
    lastAck = math.max(lastAck, stamp)
